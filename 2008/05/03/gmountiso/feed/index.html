А у меня этим занимается простецкий скрипт (depends: fuseiso):

#!/bin/bash
 
MOUNTPOINT = "/tmp/iso"
FILEMGR = "nautilus"
 
mkdir -p $MOUNTPOINT
fusermount -u $MOUNTPOINT >& /dev/null
fuseiso -c utf8 "$1" $MOUNTPOINT
$FILEMGR $MOUNTPOINT
